/* ============== Tests for env() helper ============== */

func TestEnvWithSetVariable(t *testing.T) {
	key := "TEST_INFRA_VAR"
	expected := "test_value"

	os.Setenv(key, expected)
	defer os.Unsetenv(key)

	result := env(key, "fallback")

	if result != expected {
		t.Errorf("env() = %q, want %q", result, expected)
	}
}

func TestEnvWithUnsetVariable(t *testing.T) {
	key := "NONEXISTENT_TEST_VAR_" + time.Now().Format("20060102150405")
	expected := "fallback_value"

	// Ensure the key is not set
	os.Unsetenv(key)

	result := env(key, expected)

	if result != expected {
		t.Errorf("env() = %q, want %q", result, expected)
	}
}

func TestEnvWithEmptyVariable(t *testing.T) {
	key := "TEST_INFRA_EMPTY"
	expected := "fallback_value"

	os.Setenv(key, "")
	defer os.Unsetenv(key)

	result := env(key, expected)

	if result != expected {
		t.Errorf("env() with empty string = %q, want %q (fallback)", result, expected)
	}
}

func TestEnvWithMultipleValues(t *testing.T) {
	testCases := []struct {
		name     string
		value    string
		fallback string
		expected string
	}{
		{"numeric value", "12345", "fallback", "12345"},
		{"url value", "mongodb://localhost:27017", "mongodb://default", "mongodb://localhost:27017"},
		{"empty fallback", "actual", "", "actual"},
		{"special chars", "pass@word#123", "default", "pass@word#123"},
	}

	for _, tc := range testCases {
		t.Run(tc.name, func(t *testing.T) {
			key := "TEST_VAR_" + tc.name
			os.Setenv(key, tc.value)
			defer os.Unsetenv(key)

			result := env(key, tc.fallback)
			if result != tc.expected {
				t.Errorf("env() = %q, want %q", result, tc.expected)
			}
		})
	}
}

func TestMockDatabaseIntegration(t *testing.T) {
	mockDB := &MockDatabase{}
	ctx := context.Background()

	err := mockDB.Ping(ctx)
	if err != nil {
		t.Errorf("Ping() returned unexpected error: %v", err)
	}
	if !mockDB.PingCalled {
		t.Error("Ping() was not called on mock database")
	}
}

func TestMockCacheOperations(t *testing.T) {
	mockCache := &MockCache{}
	ctx := context.Background()

	mockCache.Set(ctx, "test_key", []byte("test_value"), time.Minute)
	if mockCache.SetCalls != 1 {
		t.Errorf("SetCalls = %d, want 1", mockCache.SetCalls)
	}

	mockCache.Get(ctx, "test_key")
	if mockCache.GetCalls["test_key"] != 1 {
		t.Errorf("GetCalls['test_key'] = %d, want 1", mockCache.GetCalls["test_key"])
	}

	mockCache.Del(ctx, "test_key")
	if mockCache.DelCalls != 1 {
		t.Errorf("DelCalls = %d, want 1", mockCache.DelCalls)
	}
}

func TestMockMQOperations(t *testing.T) {
	mockMQ := &MockMQ{}
	ctx := context.Background()

	mockMQ.Publish(ctx, "test_subject", []byte("test_data"))
	if mockMQ.PublishCalls != 1 {
		t.Errorf("PublishCalls = %d, want 1", mockMQ.PublishCalls)
	}

	mockMQ.Ping(ctx)
	if !mockMQ.PingCalled {
		t.Error("Ping() was not called on mock MQ")
	}

	sub, err := mockMQ.Subscribe(ctx, "test_subject", func(ctx context.Context, msg mq.Message) error {
		return nil
	})
	if err != nil {
		t.Errorf("Subscribe() returned unexpected error: %v", err)
	}
	if mockMQ.SubscribeCalls != 1 {
		t.Errorf("SubscribeCalls = %d, want 1", mockMQ.SubscribeCalls)
	}

	if mockSub, ok := sub.(*MockSubscription); ok {
		mockSub.Unsubscribe()
		if !mockSub.UnsubscribeCalled {
			t.Error("Unsubscribe() was not called on mock subscription")
		}
	}
}

func TestEnvDefaultBehaviors(t *testing.T) {
	tests := []struct {
		name      string
		envVar    string
		envValue  string
		fallback  string
		shouldSet bool
		expected  string
	}{
		{
			name:      "mongo uri default",
			envVar:    "MONGO_URI",
			fallback:  "mongodb://localhost:27017",
			shouldSet: false,
			expected:  "mongodb://localhost:27017",
		},
		{
			name:      "redis addr default",
			envVar:    "REDIS_ADDR",
			fallback:  "localhost:6379",
			shouldSet: false,
			expected:  "localhost:6379",
		},
		{
			name:      "nats url default",
			envVar:    "NATS_URL",
			fallback:  "",
			shouldSet: false,
			expected:  "",
		},
	}

	for _, tt := range tests {
		t.Run(tt.name, func(t *testing.T) {
			if tt.shouldSet {
				os.Setenv(tt.envVar, tt.envValue)
				defer os.Unsetenv(tt.envVar)
			} else {
				os.Unsetenv(tt.envVar)
			}

			result := env(tt.envVar, tt.fallback)
			if result != tt.expected {
				t.Errorf("env(%q, %q) = %q, want %q", tt.envVar, tt.fallback, result, tt.expected)
			}
		})
	}
}
